---
title: "Поиск пользователей по имени"
api: "GET http://207.154.237.52:8000/user/name/{name}"
description: "Поиск пользователей по части имени"
---

## Параметры пути

<ParamField path="name" type="string" required>
  Строка поиска по имени пользователя (поддерживает частичное совпадение)
</ParamField>

## Ответ

<ResponseField name="total_count" type="integer">
  Общее количество найденных пользователей
</ResponseField>

<ResponseField name="user_list" type="array">
  Массив найденных пользователей

  <Expandable title="Свойства объекта">
    <ResponseField name="id" type="string">
      Уникальный идентификатор пользователя (UUID)
    </ResponseField>
    <ResponseField name="email" type="string">
      Email пользователя
    </ResponseField>
    <ResponseField name="nickname" type="string">
      Никнейм пользователя
    </ResponseField>
    <ResponseField name="username" type="string">
      Имя пользователя
    </ResponseField>
    <ResponseField name="status" type="string">
      Роль пользователя
    </ResponseField>
    <ResponseField name="user_avatar" type="string">
      Путь к аватару
    </ResponseField>
    <ResponseField name="user_banner" type="string">
      Путь к баннеру
    </ResponseField>
    <ResponseField name="password" type="string">
      Хешированный пароль (bcrypt)
    </ResponseField>
  </Expandable>
</ResponseField>

import requests

response = requests.get(
