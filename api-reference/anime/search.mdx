---
title: "Поиск аниме по названию"
api: "GET http://207.154.237.52:8000/anime/name/{name}"
description: "Поиск аниме по русскому или английскому названию"
---

## Параметры пути

<ParamField path="name" type="string" required>
  Название аниме (поддерживает частичное совпадение, кириллицу и латиницу)
</ParamField>

## Ответ

<ResponseField name="total_count" type="integer">
  Общее количество найденных аниме
</ResponseField>

<ResponseField name="anime_list" type="array">
  Массив найденных аниме

  <Expandable title="Свойства объекта">
    <ResponseField name="id" type="string">
      Внутренний UUID
    </ResponseField>
    <ResponseField name="anime_id" type="string">
      ID аниме (Shikimori)
    </ResponseField>
    <ResponseField name="russian" type="string">
      Русское название
    </ResponseField>
    <ResponseField name="english" type="string">
      Английское название
    </ResponseField>
    <ResponseField name="description" type="string">
      Описание (может содержать BBCode-разметку `[character=ID]...[/character]`)
    </ResponseField>
    <ResponseField name="score" type="number">
      Рейтинг (например, `9.23`)
    </ResponseField>
    <ResponseField name="kind" type="string">
      Тип (`tv`, `movie`, `ova`, `ona`, `special`)
    </ResponseField>
    <ResponseField name="status" type="string">
      Статус (`ongoing`, `released`, `anons`)
    </ResponseField>
    <ResponseField name="rating" type="string">
      Возрастной рейтинг (`pg_13`, `r`, `g` и т.д.)
    </ResponseField>
    <ResponseField name="season" type="string">
      Сезон выхода (например, `winter_2026`)
    </ResponseField>
    <ResponseField name="episodes" type="integer">
      Общее количество эпизодов
    </ResponseField>
    <ResponseField name="episodesAired" type="integer">
      Количество вышедших эпизодов
    </ResponseField>
    <ResponseField name="duration" type="integer">
      Длительность эпизода в минутах
    </ResponseField>
    <ResponseField name="nextEpisodeAt" type="string">
      Дата следующего эпизода (ISO 8601)
    </ResponseField>
    <ResponseField name="aired_on" type="string">
      Дата начала показа
    </ResponseField>
    <ResponseField name="released_on" type="string | null">
      Дата окончания показа (null если онгоинг)
    </ResponseField>
    <ResponseField name="poster_url" type="string">
      URL постера
    </ResponseField>
    <ResponseField name="screenshots" type="array">
      Массив URL скриншотов
    </ResponseField>
    <ResponseField name="genre_ids" type="array">
      Массив ID жанров
    </ResponseField>
    <ResponseField name="character_ids" type="array">
      Массив ID персонажей
    </ResponseField>
    <ResponseField name="related_anime_ids" type="array">
      Массив ID связанных аниме
    </ResponseField>
    <ResponseField name="related_anime_texts" type="array">
      Описания связей (например, `Предыстория`)
    </ResponseField>
    <ResponseField name="isCensored" type="boolean">
      Флаг цензуры
    </ResponseField>
    <ResponseField name="createdAt" type="string">
      Дата создания записи
    </ResponseField>
    <ResponseField name="updatedAt" type="string">
      Дата последнего обновления
    </ResponseField>
  </Expandable>
</ResponseField>

```bash cURL
  'http://207.154.237.52:8000/anime/name/%D0%A4%D1%80%D0%B8%D1%80%D0%B5%D0%BD' \
  -H 'accept: application/json'
```

```javascript JavaScript
const response = await fetch(
  `http://207.154.237.52:8000/anime/name/${encodeURIComponent("Фрирен")}`,

  }
);
const result = await response.json();
```

```python Python
import requests

response = requests.get(
    "http://207.154.237.52:8000/anime/name/Фрирен",

)

```
