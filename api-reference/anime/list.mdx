---
title: "Список аниме"
api: "GET http://207.154.237.52:8000/anime/get-anime-list"
description: "Получение списка всех аниме с пагинацией"
---

## Параметры запроса

<ParamField query="page" type="integer" default="1">
  Номер страницы
</ParamField>

<ParamField query="limit" type="integer" default="2">
  Количество записей на странице
</ParamField>

## Ответ

<ResponseField name="total_count" type="integer">
  Общее количество аниме в базе
</ResponseField>

<ResponseField name="anime_list" type="array">
  Массив аниме

  <Expandable title="Свойства объекта">
    <ResponseField name="id" type="string">
      Внутренний UUID
    </ResponseField>
    <ResponseField name="anime_id" type="string">
      ID аниме (Shikimori)
    </ResponseField>
    <ResponseField name="russian" type="string">
      Русское название
    </ResponseField>
    <ResponseField name="english" type="string">
      Английское название
    </ResponseField>
    <ResponseField name="description" type="string">
      Описание (может содержать BBCode)
    </ResponseField>
    <ResponseField name="score" type="number">
      Рейтинг
    </ResponseField>
    <ResponseField name="kind" type="string">
      Тип (`tv`, `movie`, `ova`, `ona`, `special`)
    </ResponseField>
    <ResponseField name="status" type="string">
      Статус (`ongoing`, `released`, `anons`)
    </ResponseField>
    <ResponseField name="rating" type="string">
      Возрастной рейтинг
    </ResponseField>
    <ResponseField name="season" type="string">
      Сезон выхода
    </ResponseField>
    <ResponseField name="episodes" type="integer">
      Общее количество эпизодов
    </ResponseField>
    <ResponseField name="episodesAired" type="integer">
      Вышедших эпизодов
    </ResponseField>
    <ResponseField name="duration" type="integer">
      Длительность эпизода (мин)
    </ResponseField>
    <ResponseField name="nextEpisodeAt" type="string | null">
      Дата следующего эпизода
    </ResponseField>
    <ResponseField name="aired_on" type="string">
      Дата начала показа
    </ResponseField>
    <ResponseField name="released_on" type="string | null">
      Дата окончания показа
    </ResponseField>
    <ResponseField name="poster_url" type="string">
      URL постера
    </ResponseField>
    <ResponseField name="screenshots" type="array">
      Массив URL скриншотов
    </ResponseField>
    <ResponseField name="genre_ids" type="array">
      Массив ID жанров
    </ResponseField>
    <ResponseField name="character_ids" type="array">
      Массив ID персонажей
    </ResponseField>
    <ResponseField name="related_anime_ids" type="array">
      Массив ID связанных аниме
    </ResponseField>
    <ResponseField name="related_anime_texts" type="array">
      Описания связей
    </ResponseField>
    <ResponseField name="isCensored" type="boolean">
      Флаг цензуры
    </ResponseField>
    <ResponseField name="createdAt" type="string">
      Дата создания записи
    </ResponseField>
    <ResponseField name="updatedAt" type="string">
      Дата обновления
    </ResponseField>
  </Expandable>
</ResponseField>

```python Python
import requests

response = requests.get(

)
```
