---
title: "Получить данные"
api: "GET https://api.anisign.com/v1/data"
description: "Получение списка данных"
---

## Параметры

<ParamField query="page" type="integer" default="1">
  Номер страницы для пагинации
</ParamField>

<ParamField query="limit" type="integer" default="20">
  Количество записей на странице (макс. 100)
</ParamField>

<ParamField query="search" type="string">
  Поисковый запрос для фильтрации результатов
</ParamField>

<ParamField header="Authorization" type="string" required>
  Bearer-токен для аутентификации. Пример: `Bearer ask_ваш_ключ`
</ParamField>

## Ответ

<ResponseField name="status" type="string">
  Статус ответа: `success` или `error`
</ResponseField>

<ResponseField name="data" type="array">
  Массив объектов данных

  <Expandable title="Свойства объекта">
    <ResponseField name="id" type="string">
      Уникальный идентификатор
    </ResponseField>
    <ResponseField name="name" type="string">
      Название
    </ResponseField>
    <ResponseField name="created_at" type="string">
      Дата создания в формате ISO 8601
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
```bash cURL
curl -X GET "https://api.anisign.com/v1/data?page=1&limit=20" \
  -H "Authorization: Bearer ask_ваш_ключ"
```

```javascript Node.js
const response = await fetch(
  "https://api.anisign.com/v1/data?page=1&limit=20",
  {
    headers: {
      Authorization: "Bearer ask_ваш_ключ",
    },
  }
);
const data = await response.json();
```

```python Python
import requests

response = requests.get(
    "https://api.anisign.com/v1/data",
    params={"page": 1, "limit": 20},
    headers={"Authorization": "Bearer ask_ваш_ключ"}
)
data = response.json()
```
</RequestExample>
